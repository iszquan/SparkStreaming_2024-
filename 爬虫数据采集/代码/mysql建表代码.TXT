CREATE TABLE `job_all_4_jtw` (

  `work_name` VARCHAR(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci,
  `location` VARCHAR(255),
  `salary` VARCHAR(255),
  `work_year` VARCHAR(255),
  `educate` VARCHAR(255),
  `company_name` VARCHAR(255),
  `company_type` VARCHAR(255),
  `financing_conditions` VARCHAR(255),
  `company_size` VARCHAR(255),
  `city` VARCHAR(255)
